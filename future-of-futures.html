enum Poll<T> {
    Ready(T),
    Pending
}

trait Future {
    type Output;
    fn poll(self: PinMut<Self>, cx: &mut task::Context)
        -> Poll<Self::Output>;
}
struct Ready { value: i32 }

fn ready(value: i32) → Ready {
    Ready { value }
}

impl Future for Ready {
    type Output = i32;
    fn poll(self: PinMut, cx: &mut task::Context)
        -> Poll<Self::Output>
    {
        Poll::Ready(self.value)
    }
}

Most futures won’t be ready immediately.

Instead return Poll::Pending until ready.

futures-rs to the rescue

There’s oneshot

use futures::channel::oneshot;
let (tx1, rx1) = oneshot::channel::<i32>();

let t = thread::spawn(|| tx1.send(1).unwrap());
rx1.map_ok(move |x| println!("Received: {}", x)).run_in_background();
t.join().unwrap();

And combinators on futures

let (tx1, rx) = mpsc::channel();
let tx2 = tx1.clone();
let tx3 = tx1.clone();

let fut = future::ready(1)
    .then(move |x| {
        tx1.send(x).unwrap(); // Send 1
        tx1.send(2).unwrap(); // Send 2
        future::ready(3)
    }).map(move |x| {
        tx2.send(x).unwrap(); // Send 3
        tx2.send(4).unwrap(); // Send 4
        5
    }).map(move |x| {
        tx3.send(x).unwrap(); // Send 5
    }).run_in_background();

Explicit runtime

tokio-rs to the rescue

tokio provides an async-executor and runtime

tokio::run_async(async {
    let client = Client::new();
    let uri = "http://httpbin.org/ip".parse().unwrap();

    let response = await!({
        client.get(uri).timeout(Duration::from_secs(10))
    }).unwrap();
    println!("Response: {}", response.status());

    let mut body = response.into_body();
    while let Some(chunk) = await!(body.next()) {
        let chunk = chunk.unwrap();
        println!("chunk = {}", str::from_utf8(&chunk[..]).unwrap());
    }
});
